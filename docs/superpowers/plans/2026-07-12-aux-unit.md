**Goal:** 让用户在出库/入库/盘点录入时可选用辅单位（个/条/袋/瓶/克），系统自动换算到基础单位记账。

**Tech Stack:** Flask 3.1.1 + SQLite + Jinja2 + 原生 CSS + 原生 JS；pytest；Ruff（E501/B008 忽略）。

```python
def aux_to_base(aux_qty: float, aux_rate: float) -> float:
    """辅单位 → 基础单位。aux_rate<=0 表示无辅单位，原样返回（防御）。"""
    if aux_rate <= 0:
        return float(Decimal(str(aux_qty)).quantize(Decimal('0.01')))
    return float(
        (Decimal(str(aux_qty)) / Decimal(str(aux_rate)))
        .quantize(Decimal('0.01'))
    )


def base_to_aux(base_qty: float, aux_rate: float) -> float:
    """基础单位 → 辅单位（仅用于显示换算预览）。aux_rate<=0 返回原值。"""
    if aux_rate <= 0:
        return float(Decimal(str(base_qty)).quantize(Decimal('0.01')))
    return float(
        (Decimal(str(base_qty)) * Decimal(str(aux_rate)))
        .quantize(Decimal('0.01'))
    )
```

```python
    db = get_warehouse_db()
```

```python
    items_data = db.execute("SELECT id, quantity, aux_unit, aux_rate FROM items").fetchall()

            qty = aux_to_base(qty_raw, aux_rate)
        else:

        if qty > Decimal(str(item["quantity"])):
            flash("存在出库数量大于当前库存的品项，请检查后重试")

        rows.append((int(item["id"]), qty))
    if not rows:
        flash("请至少填写一个出库数量")
        return redirect(url_for("outbound.outbound_session"))
    from blueprints.procurement import mark_procurement_invalid
    for item_id, qty in rows:
        cur = db.execute(
            """INSERT INTO outbound_requests
               (item_id, requested_quantity, reason, status, rolled_back, created_at)

        )
        mark_procurement_invalid(item_id)
```

```javascript
    var rows = document.querySelectorAll('.entry-row');
    for (var i = 0; i < rows.length; i++) {
      (function (row) {
        var seg = row.querySelector('.seg');
        if (!seg) return;
        var rate = parseFloat(seg.getAttribute('data-rate') || '0');
        var baseUnit = row.querySelector('input[data-base-unit]');
        var baseName = baseUnit ? baseUnit.getAttribute('data-base-unit') : '';
        var auxName = baseUnit ? baseUnit.getAttribute('data-aux-unit') : '';
        var input = row.querySelector('input[type=number]');
        var hidden = row.querySelector('input[type=hidden]');
        var label = row.querySelector('.er-unit');
        var hint = row.querySelector('.calc-hint');
        var btns = seg.querySelectorAll('button');
        for (var j = 0; j < btns.length; j++) {
          btns[j].addEventListener('click', function () {
            for (var k = 0; k < btns.length; k++) btns[k].classList.remove('active');
            this.classList.add('active');
            var u = this.getAttribute('data-u');
            hidden.value = u;
            if (u === 'aux') {
              label.textContent = auxName;
              input.setAttribute('placeholder', '0 ' + auxName);
            } else {
              label.textContent = baseName;
              input.setAttribute('placeholder', '0');
            }
            updateHint();
          });
        }
        function updateHint() {
          var v = parseFloat(input.value || '0');
          if (isNaN(v) || v === 0) { if (hint) hint.setAttribute('hidden', ''); return; }
          if (hidden.value === 'aux' && rate > 0) {
            var base = Math.round((v / rate) * 100) / 100;
            if (hint) {
              hint.textContent = v + ' ' + auxName + ' ≈ ' + base + ' ' + baseName;
              hint.removeAttribute('hidden');
            }
          } else {
            if (hint) hint.setAttribute('hidden', '');
          }
        }
        input.addEventListener('input', updateHint);
      })(rows[i]);
    }
  })();
```

```javascript
  (function () {
    var rows = document.querySelectorAll('.entry-row');
    for (var i = 0; i < rows.length; i++) {
      (function (row) {
        var seg = row.querySelector('.seg');
        if (!seg) return;
        var rate = parseFloat(seg.getAttribute('data-rate') || '0');
        var baseUnit = row.querySelector('input[data-base-unit]');
        var baseName = baseUnit ? baseUnit.getAttribute('data-base-unit') : '';
        var auxName = baseUnit ? baseUnit.getAttribute('data-aux-unit') : '';
        var input = row.querySelector('input[type=number]');
        var hidden = row.querySelector('input[type=hidden]');
        var label = row.querySelector('.er-unit');
        var hint = row.querySelector('.calc-hint');
        var btns = seg.querySelectorAll('button');
        for (var j = 0; j < btns.length; j++) {
          btns[j].addEventListener('click', function () {
            for (var k = 0; k < btns.length; k++) btns[k].classList.remove('active');
            this.classList.add('active');
            var u = this.getAttribute('data-u');
            hidden.value = u;
            if (u === 'aux') {
              label.textContent = auxName;
              input.setAttribute('placeholder', '0 ' + auxName);
            } else {
              label.textContent = baseName;
              input.setAttribute('placeholder', '0');
            }
            updateHint();
          });
        }
        function updateHint() {
          var v = parseFloat(input.value || '0');
          if (isNaN(v) || v === 0) { if (hint) hint.setAttribute('hidden', ''); return; }
          if (hidden.value === 'aux' && rate > 0) {
            var base = Math.round((v / rate) * 100) / 100;
            if (hint) {
              hint.textContent = v + ' ' + auxName + ' ≈ ' + base + ' ' + baseName;
              hint.removeAttribute('hidden');
            }
          } else {
            if (hint) hint.setAttribute('hidden', '');
          }
        }
        input.addEventListener('input', updateHint);
      })(rows[i]);
    }
  })();
```

```python
@bp.route("/stocktake/submit", methods=["POST"])
@require_login
def stocktake_submit():
    db = get_warehouse_db()
    note = request.form.get("note", "").strip()
    items_data = db.execute("SELECT id, quantity, aux_unit, aux_rate FROM items").fetchall()
    from ._helpers import aux_to_base

    changed_rows = []
    for item in items_data:
        raw = request.form.get(f"actual_{item['id']}", "").strip()
        if raw == "":
            continue
        qty_raw = parse_qty(raw)
        unit_choice = request.form.get(f"actual_{item['id']}_unit", "base")
        if unit_choice == "aux":
            aux_rate = float(item["aux_rate"] or 0)
            if aux_rate <= 0:
                flash("存在品项未启用辅单位，请用基础单位录入")
                return redirect(url_for("stocktake.stocktake_session"))
            actual_quantity = aux_to_base(qty_raw, aux_rate)
        else:
            actual_quantity = qty_raw
        previous_quantity = parse_qty(item["quantity"])
        diff = actual_quantity - previous_quantity
        changed_rows.append((int(item["id"]), previous_quantity, actual_quantity, diff))

    if not changed_rows:
        flash("请至少填写一个盘点数量")
        return redirect(url_for("stocktake.stocktake_session"))

    cur = db.execute(
        "INSERT INTO stocktake_batches (created_at, note, status, rolled_back) VALUES (?, ?, 'pending', 0)",
        (now(), note),
    )
    batch_id = cur.lastrowid
    for item_id, previous_quantity, actual_quantity, diff in changed_rows:
        db.execute(
            """INSERT INTO stocktakes
               (item_id, previous_quantity, actual_quantity, diff, batch_id, created_at, note)
               VALUES (?, ?, ?, ?, ?, ?, ?)""",
            (item_id, previous_quantity, actual_quantity, diff, batch_id, now(), note),
        )
    db.commit()
    audit("stocktake.submit", "batch", batch_id, {"count": len(changed_rows)})
    flash(f"盘点批次 #{batch_id} 已提交，等待审核")
    return redirect(url_for("stocktake.stocktake_list"))
```
